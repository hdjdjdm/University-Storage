/* 2.1.4.Напишите функцию static int Cut(int n), которая по данному натуральному числу n
возвращает число, полученное отбрасыванием старшего и младшего разрядов (например, при вводе 179
программа выводит 7). Определите время работы функции для 1000000 повторов. */
using System;
using System.Diagnostics;

namespace Lr2
{
    class Num1
    {
        static int Cut(int n)
        {
            //Убираем последний разряд
            n /= 10;

            //Убираем первый разряд
            int x = 1;
            while (x < n)
            {
                x *= 10;
            }
            n %= x/10;
            
            return n;
        }

        static void Main()
        {
            //Вариант
            int variant = (14 - 1) % 5 + 1; // = 4
            Console.WriteLine($"Вариант {variant}\n");

            //Ввод числа
            int n;
            do
            {
                Console.Write("Введите натуральное число (больше двух цифр): ");
                n = Convert.ToInt32(Console.ReadLine());
            } while (n < 100);
            Console.WriteLine();

            //Время работы функции для 1000000 повторов
            int i = 0;
            var time = Stopwatch.StartNew();
            while (i < 1000000)
            {
                Cut(n);
                i++;
            }
            time.Stop();
            double t = time.ElapsedMilliseconds;

            //Результат
            Console.WriteLine($"Результат: {Cut(n)}");
            Console.WriteLine($"Время выполнения функции Cut 1000000 раз = {t}ms");
            Console.ReadKey();

        }
    }
}
